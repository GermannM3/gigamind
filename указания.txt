```python
import sqlite3
from sentence_transformers import SentenceTransformer
import faiss
import numpy as np
import os

class GigaMemory:
    def __init__(self, db_path="data/memory.db"):
        os.makedirs("data", exist_ok=True)
        self.conn = sqlite3.connect(db_path)
        self.create_table()
        self.model = SentenceTransformer('all-MiniLM-L6-v2')
        self.index = faiss.IndexFlatL2(384)  # –†–∞–∑–º–µ—Ä —ç–º–±–µ–¥–¥–∏–Ω–≥–∞ –º–æ–¥–µ–ª–∏
        self.history = []

    def create_table(self):
        self.conn.execute('''
            CREATE TABLE IF NOT EXISTS memory (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                user_id TEXT,
                timestamp DATETIME DEFAULT CURRENT_TIMESTAMP,
                query TEXT,
                response TEXT,
                tags TEXT
            )
        ''')
        self.conn.commit()

    def add_memory(self, user_id, query, response, tags=""):
        self.conn.execute(
            "INSERT INTO memory (user_id, query, response, tags) VALUES (?, ?, ?, ?)",
            (user_id, query, response, tags)
        )
        self.conn.commit()
        # –î–æ–±–∞–≤–ª—è–µ–º –≤ FAISS
        embedding = self.model.encode([query])[0]
        self.index.add(np.array([embedding]))
        self.history.append((user_id, query, response, tags))

    def search_memory(self, query, top_k=3):
        embedding = self.model.encode([query])[0]
        D, I = self.index.search(np.array([embedding]), top_k)
        results = []
        for idx in I[0]:
            if idx < len(self.history):
                results.append(self.history[idx])
        return results
```

from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer

class TinyJudge:
    def __init__(self):
        self.analyzer = SentimentIntensityAnalyzer()

    def evaluate(self, query, response, memory_context=""):
        # –ü—Ä–æ—Å—Ç–∞—è —ç–≤—Ä–∏—Å—Ç–∏–∫–∞: –æ—Ü–µ–Ω–∏–≤–∞–µ–º —Ç–æ–Ω–∞–ª—å–Ω–æ—Å—Ç—å –∏ –¥–ª–∏–Ω—É
        sentiment = self.analyzer.polarity_scores(response)['compound']
        relevance = 5 if len(response) > 10 else 3
        empathy = 5 if sentiment > 0.1 else 3
        usefulness = 5 if "://" not in response else 4  # –µ—Å–ª–∏ —Å—Å—ã–ª–∫–∞ ‚Äî –ø–æ–ª–µ–∑–Ω–æ

        score = round((relevance + empathy + usefulness) / 3, 1)
        return score, f"Sentiment: {sentiment:.2f}, Relevance: {relevance}, Empathy: {empathy}"
